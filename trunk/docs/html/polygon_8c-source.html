00013 /*!\fn float polygon_area(polygon *p)
00014 Esta funcion calcula el area de un poligono simple, el algoritmo fue tomado
00015 de Graphic Gems II, del articulo The Area of a Simple Polygon.
00016 
00017 \param [in] p Poligono simple
00018 \return El area del poligono
00019 */
00020 float polygon_area(polygon *p)
00021 {
00022     unsigned int i;
00023     float area = 0;
00024     for (i=0;i<p->nvertices;i++)
00025     {
00026         area += p->v[i].x * p->v[(i + 1) % p->nvertices].y;
00027         area -= p->v[i].y * p->v[(i + 1) % p->nvertices].x;
00028     }
00029     area /= 2;
00030 
00031     return(area < 0 ? -area : area);

00107 La funcion rota el poligono p un angulo t, teniendo como eje de rotacion el

00111 ademas que evita exageradas translaciones relativas del poligono al ser rotado.

00113 \param [in,out] p Poligono simple
00114 \param [in] t Angulo en radianes a rotar la figura en sentido antihorario.
00115 */

00118     int i;
00119     point pc;
00120 
00121     float sent = sin(t);
00122     float cost = cos(t);

00125
